    ```python
    """OpenAI AI 프로바이더 구현."""
    from __future__ import annotations
    from openai import OpenAI
    from meeting_transcriber.ai.provider_base import AIProvider
    from meeting_transcriber.utils.keychain import get_api_key

    class OpenAIProvider(AIProvider):
        """OpenAI GPT를 사용하는 AI 프로바이더."""
        def __init__(self, api_key: str | None = None, model: str = "gpt-4o-mini") -> None:
            key = api_key or get_api_key("openai")
            if not key:
                raise ValueError("OpenAI API key not found. Set it in Settings > API Keys.")
            self._client = OpenAI(api_key=key)
            self._model = model
        def _call(self, prompt: str) -> str:
            response = self._client.chat.completions.create(
                model=self._model,
                messages=[{"role": "user", "content": prompt}],
            )
            return response.choices[0].message.content.strip()
    ```
    - Implement all 5 ABC methods using the SAME prompts as GeminiProvider (copy the prompt strings exactly)
    - Use `gpt-4o-mini` as default model (cost-effective per research recommendation)

    ```python
    """AI 프로바이더 관리 및 폴백 처리."""
    from __future__ import annotations
    import logging
    from typing import Any
    from meeting_transcriber.ai.provider_base import AIProvider
    from meeting_transcriber.utils.keychain import get_api_key

    logger = logging.getLogger(__name__)

    class ProviderManager:
        """AI 프로바이더 체인을 관리하고 폴백을 처리한다."""

        PROVIDERS: dict[str, tuple[str, str]] = {
            "gemini": ("meeting_transcriber.ai.gemini_provider", "GeminiProvider"),
            "openai": ("meeting_transcriber.ai.openai_provider", "OpenAIProvider"),
            "anthropic": ("meeting_transcriber.ai.anthropic_provider", "AnthropicProvider"),
        }
    ```

    Methods to implement:

    1. `get_provider_chain(self, settings: dict[str, Any]) -> list[AIProvider]` — per D-15:
       - Read `settings["ai"]["default_provider"]` (default "gemini")
       - Build ordered list: [default, ...others that have keys]
       - Instantiate each using lazy import from PROVIDERS dict
       - Skip providers without keys (get_api_key returns None)

    2. `get_provider_for_task(self, task: str, settings: dict[str, Any]) -> list[AIProvider]`:
       - Check `settings["ai"]["task_overrides"]` for task-specific provider
       - If override exists, use that provider as default for this chain
       - Otherwise delegate to get_provider_chain

    3. `execute_with_fallback(self, chain: list[AIProvider], method: str, *args: Any, **kwargs: Any) -> tuple[Any, str | None]` — per D-13, D-14:
       - Try each provider in order
       - On success: return (result, None) if first provider, (result, "{Failed} failed, using {Success}") if fallback
       - On all fail: raise RuntimeError(f"All providers failed. Last error: {last_error}")
       - Log each failure with logger.warning

    4. `_instantiate(self, name: str) -> AIProvider`:
       - Lazy import using importlib from PROVIDERS dict
       - Return provider instance (key loaded from keychain automatically in provider __init__)

    5. `_has_key(self, name: str) -> bool`:
       - Return `get_api_key(name) is not None`
